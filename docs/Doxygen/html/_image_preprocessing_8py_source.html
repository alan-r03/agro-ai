28 def getAdvancedFeatures(imageIn):
29 """
30 Returns a tuple of advanced features.
31
32 Parameters
33 ----------
34 imageIn : Image
35 The image to process.
36
37 Returns
38 -------
39 returnValues : tuple
40 numbers.
41
42 """
43 lowRed = 165
44 highRed = 240
45 lowGreen = 160
46 highGreen = 200
47 lowBlue = 135
48 highBlue = 240
49
50 rgb_img = imageIn
51 red = rgb_img[:, :, 0]
52 hsv_img = rgb2hsv(rgb_img)
53 hue_img = hsv_img[:, :, 0]
54 sat_img = hsv_img[:, :, 1]
55 value_img = hsv_img[:, :, 2]
56
57 #saturation mask to isolate foreground
58 satMask = (sat_img > .11) | (value_img > .3)
59 #hue and value mask to remove additional brown from background
60 mask = (hue_img > .14) | (value_img > .48)
61 #healthy corn mask to remove healthy corn, leaving only blighted pixels
62 nonBlightMask = hue_img < .14
63 #get foreground
64 rawForeground = np.zeros_like(rgb_img)
65 rawForeground[mask] = rgb_img[mask]
66 #reduce brown in background
67 foreground = np.zeros_like(rgb_img)
68 foreground[satMask] = rawForeground[satMask]
69 #get blighted pixels from foreground
70 blightedPixels = np.zeros_like(rgb_img)
71 blightedPixels[nonBlightMask] = foreground[nonBlightMask]
72 #combine into one band
73 blightedHSV = np.bitwise_or(blightedPixels[:,:,0], blightedPixels[:,:,1])
74 blightedHSV = np.bitwise_or(blightedHSV, blightedPixels[:,:,2])
75
76 red = rgb_img[:, :, 0]
77 green = rgb_img[:, :, 1]
78 blue = rgb_img [:, :, 2]
79 binary_green = lowGreen < green
80 binary_blue = lowBlue < blue
81 binary_red = lowRed < red
82 RGB_Blights = np.bitwise_and(binary_red, binary_green)
83 #'brown' pixels within each RGB threshold
84 RGB_Blights = np.bitwise_and(RGB_Blights, binary_blue)
85 HSV_and_RGB = np.bitwise_and(RGB_Blights, blightedHSV)
86 #get features
87 numForegroundPixels = np.count_nonzero(foreground)
88 numBlightedHSVPixels = np.count_nonzero(blightedHSV)
89 blightedHSVRatio = numBlightedHSVPixels / numForegroundPixels
90 num_RGB_blightedPixels = np.count_nonzero(RGB_Blights)
91 blightedRGBRatio = num_RGB_blightedPixels / numForegroundPixels
92 numBlightedBothPixels = np.count_nonzero(HSV_and_RGB)
93 blightedBothRatio = numBlightedBothPixels / numForegroundPixels
94 returnValues = (numForegroundPixels, numBlightedHSVPixels, blightedHSVRatio, num_RGB_blightedPixels,
95 blightedRGBRatio, numBlightedBothPixels, blightedBothRatio)
96
97 return returnValues

142 """

157 image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

187 def FdHistogram(image, mask=None, bins = 8):
188 """
189 Extracts color histogram feature from an image
190 Parameters
191 ----------
192
193 image : imread
194 The image used for feature extraction
195 Returns
196 -------
197 Feature : Float Array
198 The color histogram in the image.
199 Reference
200 ---------
201 https://gogul.dev/software/image-classification-python
202 """
203 # convert the image to HSV color-space
204 image = cv2.cvtColor(image, cv2.COLOR_BGR2HSV)
205 # compute the color histogram
206 hist = cv2.calcHist([image], [0, 1, 2], None, [bins, bins, bins], [0, 256, 0, 256, 0, 256])
207 # normalize the histogram
208 cv2.normalize(hist, hist)
209 # return the histogram
210 return hist.flatten()
